#下载头像并保存在/avatar/{时间戳}.jpg中
import requests
import os
import time
i = 0
c = os.path.dirname(__file__)
url = 'https://api.yimian.xyz/img?type=head'
while True:
    x = requests.get(url)
    b = int(time.time())
    a = os.path.join('avatar','{b}.jpg'.format(b=b))
    with open(a,'wb') as f:
        f.write(x.content)
    print(b,'.jpg')
